module.exports.find = (id) ->
  id = parseInt(id, 10) 

  for product in products 
    if product.id is id then return product 



module.exports.set = (id, product) ->
  id = parseInt(id, 10) 

  product.id = id
  products[id - 1] = product



module.exports.new = () ->
  name : ''
  description: ''
  price: 0



module.exports.insert = (product) ->
  id = products.length + 1
  product.id = id
  products[id - 1] = product
