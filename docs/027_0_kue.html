sending email to see the progress

 
sequence = 0

setInterval ->
  sequence += 1  
  ((sequence) ->
    job = jobs.create 'email',
      title: 'Hello',
      to: 'nodetuts_in_coffeescript@iscool.com',
      body: 'Hello from Node Tuts!'
    .attempts(5).priority('high').save()
